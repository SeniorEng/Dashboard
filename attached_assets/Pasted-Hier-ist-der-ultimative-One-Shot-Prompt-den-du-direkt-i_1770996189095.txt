#!/bin/bash
echo "🚀 INITIALISIERE TEAM-AUDIT..."

echo "\n🔍 1. CODE QUALITY (Linter/Typecheck)"
# Passe dies an deine tatsächlichen Befehle an, falls abweichend
npm run typecheck || echo "⚠️ TypeScript Fehler gefunden!"
npm run lint || echo "⚠️ Linting Fehler gefunden!"

echo "\n🛡️ 2. SECURITY CHECK"
npm audit --audit-level=high

echo "\n📝 3. TODO SCAN (Offene Aufgaben)"
grep -r "TODO" ./src --exclude-dir=node_modules | head -n 5

echo "\n✅ AUTOMATISCHER TEIL ABGESCHLOSSEN."
echo "Bitte jetzt den AI Architect um ein inhaltliches Review bitten: '/audit'"
